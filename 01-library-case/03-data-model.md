# Кейс: Библиотека — Модель данных

## ER-диаграмма (Mermaid)

```mermaid
erDiagram
    Книги {
        int id PK "уникальный идентификатор"
        string название "название книги"
        string автор "автор"
        string статус "в_наличии/на_руках/забронирована"
    }

    Читатели {
        int id PK "уникальный идентификатор"
        string полное_имя "ФИО"
        string номер_билета "номер читательского билета"
    }

    Бронирования {
        int id PK "уникальный идентификатор"
        int книга_id FK "ссылка на книгу"
        int читатель_id FK "ссылка на читателя"
        date дата_брони "дата бронирования"
        date срок_до "дата окончания брони"
    }

    Книги ||--o{ Бронирования : "бронируется"
    Читатели ||--o{ Бронирования : "совершает"
